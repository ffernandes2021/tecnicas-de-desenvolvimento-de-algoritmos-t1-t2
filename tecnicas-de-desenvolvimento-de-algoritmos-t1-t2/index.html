  • Se i = 2 escrever os três valores a, b, c em ordem decrescente.

    se (a > b) e (a > c) entao
        se b > c entao
            maior <- a
            medio <- b
            menor <- c
        senao
            maior <- a
            medio <- c
            menor <- b
        fimse
    senao
        se (b > a) e (b > c) entao
            se a > c entao
                maior <- b
                medio <- a
                menor <- c
            senao
                maior <- b
                medio <- c
                menor <- a
            fimse

            se (c > a) e (c > b) entao

                    medio <- a
                    menor <- b

                    maior <- c
                    medio <- b
                    menor <- a

            escreva("Os números em ordem crescente: ",menor,medio,maior)
